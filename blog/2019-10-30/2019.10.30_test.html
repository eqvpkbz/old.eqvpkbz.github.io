#include<cstdio>
#include<algorithm>
#include<cstdlib>
using namespace std;
#define file(SSS) freopen(SSS".in","r",stdin),freopen(SSS".out","w",stdout)
#define file_close fclose(stdin),fclose(stdout)
const int Maxn = 200009;
const long long INF = 2147483647;
struct seg {int l, r;}a[Maxn];
int n,x,y;
bool cmp(const seg u, const seg v){return u.r < v.r;}
int main() 
{
    file("x");
    scanf("%d",&n);
    for(int i = 1;i <= n;i ++)
    {
        scanf("%d%d",&x,&y);
        a[i] = (seg){x - y, x + y};
    }
    sort(a + 1, a + n + 1,cmp);
    int ans = 0, lef = -INF;
    for(int i = 1;i <= n;i ++)
        if (lef < a[i].l) lef = a[i].r, ++ ans;
    printf("%d",ans);
    file_close;
    return 0;
}
